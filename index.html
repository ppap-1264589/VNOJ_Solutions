#include <bits/stdc++.h>
#define Task "MST"
#define pii pair<int, int>
#define f first
#define s second
#define up(i,a,b) for (int i = (a); i <= b; i++)
#define SOME -1
#define NONE 0
#define ANY 1
using namespace std;

const int maxn = 100001;
int n,m;
vector<pii> a[maxn];
struct edge{
    int u,v,w,id;
} E[maxn+1];

bool cmp(const edge& x, const edge& y){
    return (x.w < y.w);
}

void input(){
    cin >> n >> m;
    up(i,1,m){
        cin >> E[i].u >> E[i].v >> E[i].w;
        E[i].id = i;
    }
}

int num[maxn], low[maxn];
int tdfs;
vector<edge> pend;
vector<pii> G[maxn];
int par[maxn];
int type[maxn];

void Tarjan(int u, int uid){
    num[u] = low[u] = ++tdfs;
    for (pii x : G[u]){
        int v = x.f;
        int vid = x.s;
        if (vid == uid) continue;
        if (!num[v]){
            Tarjan(v, vid);
            low[u] = min(low[u], low[v]);
            if (low[v] == num[v]){
                type[vid] = ANY;
            }
        }
        else low[u] = min(low[u], num[v]);
    }
}

int root(int u){
    if (par[u] == 0) return u;
    int x = root(par[u]);
    par[u] = x;
    return x;
}

void Union(int u, int v){
    u = root(u);
    v = root(v);
    if (u != v) par[u] = v;
}

void Compress_Graph(){
    for (int i = 0; i < (int)pend.size(); i++){
        int u = pend[i].u;
        int v = pend[i].v;
        u = root(u);
        v = root(v);
        pend[i].u = u;
        pend[i].v = v;

        G[u].clear();
        G[v].clear();
        num[u] = num[v] = 0;
    }

    for (auto e : pend){
        int u = e.u;
        int v = e.v;
        int id = e.id;
        if (u == v) type[id] = NONE;
        else {
            type[id] = SOME;
            G[u].push_back({v, id});
            G[v].push_back({u, id});
        }
    }

    for (auto e : pend) if (num[e.u] == 0) Tarjan(e.u, -1);
    for (auto e : pend) Union(e.u, e.v);
}

void Improved_Kruskal(){
    sort(E+1, E+m+1, cmp);
    for (auto e : E){
        if (!pend.empty() && pend.back().w == e.w){
            pend.push_back(e);
        }
        else {
            Compress_Graph();
            pend.clear();
            pend.push_back(e);
        }
    }
    if (!pend.empty()) Compress_Graph();
}

void output(){
    up(i, 1, m){
        if (type[i] == NONE) cout << "none";
        if (type[i] == ANY) cout << "any";
        if (type[i] == SOME) cout << "at least one";
        cout << "\n";
    }
}

signed main(){
    ios_base::sync_with_stdio(false);
    cin.tie(0);
    cout.tie(0);

    input();    Improved_Kruskal();    output();
}
